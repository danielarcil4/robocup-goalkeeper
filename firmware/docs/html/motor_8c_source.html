22bool motor_init(motor_brushless_t *motor) {
23 if (!motor || motor->max_speed_percent > 100) {
24 ESP_LOGE(TAG, "Invalid motor configuration");
25 return false;
26 }
27
28 ledc_timer_config_t timer_conf = {
29 .duty_resolution = motor->resolution_bits,
30 .freq_hz = 60, // 60 recommended for motors
31 .speed_mode = motor->speed_mode,
32 .timer_num = motor->timer_num,
33 .clk_cfg = LEDC_AUTO_CLK
34 };
35 // Configure timer only if not already configured for this timer_num and speed_mode
36 static bool timer_initialized[LEDC_TIMER_MAX][2] = { false };
37 int mode_idx = (motor->speed_mode == LEDC_LOW_SPEED_MODE) ? 1 : 0;
38 if (!timer_initialized[motor->timer_num][mode_idx]) {
39 if (ledc_timer_config(&timer_conf) != ESP_OK) {
40 ESP_LOGE(TAG, "Failed to configure LEDC timer");
41 return false;
42 }
43 timer_initialized[motor->timer_num][mode_idx] = true;
44 }
45
46 ledc_channel_config_t speed_channel = {
47 .channel = motor->speed_channel,
48 .duty = (uint32_t)(motor->min_speed_percent * ((1 << motor->resolution_bits) - 1) /100), // Set initial duty top of MOTOR_MAX_SPEED_PERCENT
49 .gpio_num = motor->pwm_pin_speed,
50 .speed_mode = motor->speed_mode,
51 .hpoint = 0,
52 .timer_sel = motor->timer_num
53 };
54 if (ledc_channel_config(&speed_channel) != ESP_OK) {
55 ESP_LOGE(TAG, "Failed to configure speed channel");
56 return false;

59 ledc_channel_config_t reverse_channel = {

62 .gpio_num = motor->pwm_pin_reverse,
63 .speed_mode = motor->speed_mode,

65 .timer_sel = motor->timer_num

82void motor_set_speed(motor_brushless_t *motor, float signed_speed_percent)
83{
84 if (!motor) return;
85
86 // Clamp input to [-100.0, 100.0]
87 signed_speed_percent = fmaxf(fminf(signed_speed_percent, 100.0f), -100.0f);
88
89 // Determine direction and absolute speed
90 bool reverse = (signed_speed_percent < 0.0f);
91 float abs_speed_percent = fabsf(signed_speed_percent);
92
93 // Scale to effective speed using configured min/max bounds
94 float speed_range = motor->max_speed_percent - motor->min_speed_percent;
95 float effective_speed_percent = motor->min_speed_percent + (abs_speed_percent * speed_range / 100.0f);
96
97 // Calculate duty cycle from effective speed
98 uint32_t max_duty = (1U << motor->resolution_bits) - 1U;
99 uint32_t duty = (uint32_t)lroundf((effective_speed_percent / 100.0f) * max_duty);
100
101 // Log values
102 // ESP_LOGI(TAG, "Set motor speed: %.2f%% | Dir: %s | Effective: %.2f%% | Duty: %lu",
103 // signed_speed_percent, reverse ? "REVERSE" : "FORWARD", effective_speed_percent, (unsigned long)duty);
104
105 // Apply speed duty
106 ledc_set_duty(motor->speed_mode, motor->speed_channel, duty);
107 ledc_update_duty(motor->speed_mode, motor->speed_channel);
108
109 // Set reverse PWM signal: 25% for reverse ON, 75% for OFF
110 uint32_t rev_duty = reverse ? (max_duty * 25U) / 100U : (max_duty * 75U) / 100U;
111 ledc_set_duty(motor->speed_mode, motor->reverse_channel, rev_duty);
112 ledc_update_duty(motor->speed_mode, motor->reverse_channel);
113
114 // Update state
115 motor->is_reversed = reverse;
116}
