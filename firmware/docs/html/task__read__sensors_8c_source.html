73static inline float kalman_update(Kalman1D *kf, float measurement) {
74 kf->P += kf->Q;
75 float K = kf->P / (kf->P + kf->R);
76 kf->x += K * (measurement - kf->x);
77 kf->P *= (1.0f - K);
78 return kf->x;
79}

92static inline float compute_angular_velocity(angular_velocity_t *sensor, float angle_deg, int64_t time_us)
93{
94 float delta_deg = angle_deg - sensor->last_angle_deg;
95
96 // Handle wrap-around (360 -> 0 or 0 -> 360)
97 if (delta_deg > 180.0f) delta_deg -= 360.0f;
98 else if (delta_deg < -180.0f) delta_deg += 360.0f;
99
100 int64_t delta_time_us = time_us - sensor->last_time_us;
101 if (delta_time_us <= 0) return 0.0f;
102
103 float deg_per_sec = delta_deg * (1e6f / (float)delta_time_us);
104 float rad_per_sec = deg_per_sec * (M_PI / 180.0f);
105
106 sensor->last_angle_deg = angle_deg;
107 sensor->last_time_us = time_us;
108
109 return rad_per_sec;
110}
