23static uart_port_t find_uart_num(uint8_t gpio_tx, uint8_t gpio_rx, uint8_t gpio_rts, uint8_t gpio_cts)
24{
25 // UART number for the ESP32-S3 microcontroller
26 if (gpio_tx == 43 && gpio_rx == 44)
27 {
28 return UART_NUM_0;
29 }
30 else if (gpio_tx == 17 && gpio_rx == 18)
31 {
32 return UART_NUM_1;
33 }
34
35 return UART_NUM_2;
36}

41int uart_init(uart_t *uart_config, uint32_t baud_rate, uint16_t buffer_size, int8_t gpio_tx, int8_t gpio_rx, int8_t gpio_rts, int8_t gpio_cts)
42{
43 // Check if the UART gpio pins are valid
44 if (gpio_tx == UART_PIN_NO_USE || gpio_rx == UART_PIN_NO_USE)
45 {
46 ESP_LOGE("UART_INIT", "Invalid UART TX or RX pin");
47 return -1;
48 }
49
50
51
52 // Find UART number for the ESP32-S3 microcontroller
53 uart_port_t uart_num = find_uart_num(gpio_tx, gpio_rx, gpio_rts, gpio_cts);
54
55 // Configure UART parameters
56 uart_config->baud_rate = baud_rate;
57 uart_config->buffer_size = buffer_size;
58 uart_config->uart_num = uart_num;
59 uart_config->gpio_tx = gpio_tx;
60 uart_config->gpio_rx = gpio_rx;
61 uart_config->gpio_rts = gpio_rts;
62 uart_config->gpio_cts = gpio_cts;
63
64 // Configure UART structure
65 uart_config->uart_config = (uart_config_t){
66 .baud_rate = baud_rate,
67 .data_bits = UART_DATA_8_BITS,
68 .parity = UART_PARITY_DISABLE,
69 .stop_bits = UART_STOP_BITS_1,
70 .flow_ctrl = (gpio_rts != UART_PIN_NO_USE && gpio_cts != UART_PIN_NO_USE)
71 ? UART_HW_FLOWCTRL_CTS_RTS
72 : UART_HW_FLOWCTRL_DISABLE,
73 .source_clk = UART_SCLK_DEFAULT,
74 };
75
76
77
78
79 int log = 0;
80 // Configure UART parameters and check if UART was initialized successfully
81 if (uart_param_config(uart_num, &(uart_config->uart_config)) != ESP_OK)
82 {
83 ESP_LOGE("UART_INIT", "Failed to configure UART parameters");
84 log = -1;
85 }
86
87 // Configure UART pins
88 if (uart_set_pin(uart_num, gpio_tx, gpio_rx, gpio_rts, gpio_cts) != ESP_OK)
89 {
90 ESP_LOGE("UART_INIT", "Failed to configure UART pins");
91 log = -1;
92 }
93
94 // Install UART driver
95 if (uart_driver_install(uart_num, buffer_size * 2, 0, 0, NULL, 0) != ESP_OK)
96 {
97 ESP_LOGE("UART_INIT", "Failed to install UART driver");
98 log = -1;
99 }
100
101 // Clear UART buffer
102 if (uart_flush(uart_num) != ESP_OK)
103 {
104 ESP_LOGE("UART_INIT", "Failed to flush UART buffer");
105 log = -1;
106 }
107
108 // ESP_LOGI("UART_INIT", "UART initialized successfully");
109 return log;
110}

308{
309
310 esp_err_t ret = ESP_OK;
311
312 adc->gpio_out = gpio_out;
313
314 // From GPIO to ADC channel

316 ESP_GOTO_ON_ERROR(adc_oneshot_io_to_channel(adc->gpio_out, &adc->unit, &adc->chan), err, TAG_ADC, "adc io to channel failed");
317 ESP_LOGI(TAG_ADC, "ADC channel: %d", adc->chan);
318
319 // ------------- ADC pin OUT configuration ------------- //
320 // The DIG ADC2 controller of ESP32-S3 doesn’t work properly (pag. 1444).

326 .unit_id = ADC_CONF_UNIT,

329
330 // ADC Config

332 .atten = ADC_ATTEN,
333 .bitwidth = ADC_BIT_WIDTH,
334 };
335 ESP_GOTO_ON_ERROR(adc_oneshot_config_channel(adc_handle, adc->chan, &config), err, TAG_ADC, "adc config failed");
336
337 // ADC calibration
338 adc_cali_handle_t cali_handle = NULL;
339 adc_cali_curve_fitting_config_t cali_config = {
340 .unit_id = ADC_CONF_UNIT,
341 .chan = adc->chan,
342 .atten = ADC_ATTEN,
343 .bitwidth = ADC_BIT_WIDTH,
344 };
345 adc->is_calibrated = false;
346 ESP_GOTO_ON_ERROR(adc_cali_create_scheme_curve_fitting(&cali_config, &cali_handle), err, TAG_ADC, "adc calibration failed");
347
348 adc->is_calibrated = true;
