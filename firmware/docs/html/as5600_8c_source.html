41float AS5600_ADC_GetAngle(AS5600_t *as5600)
42{
43 float angle;
44 if (as5600->adc_handle.is_calibrated && as5600->conf.OUTS == AS5600_OUTPUT_STAGE_ANALOG_RR) {
45 uint16_t voltage;
46 adc_read_mvolt(&as5600->adc_handle, &voltage);
47 voltage = LIMIT(voltage, VCC_3V3_MIN_RR_MV, VCC_3V3_MAX_RR_MV); // The OUT pin of the AS5600 sensor has a range of 10%-90% of VCC
48 angle = MAP((float)voltage, VCC_3V3_MIN_RR_MV, VCC_3V3_MAX_RR_MV, 0, 360); // Map the voltage to the angle
49 }
50 else {
51 angle = -1;
52 }
53 return angle;
54}
