87void vTaskMove(void* arg)
88{
89 TickType_t xLastWakeTime = xTaskGetTickCount();
90 float t = 0.0f;
91 float prev_angle = 0.0f;
92 Velocity speed_estimated = {0};
93
94 while (1) {
95 // Calculate the command velocities based on a circular trajectory
96 float vx_cmd = -CIRCULAR_RADIUS * OMEGA_CIRC * sinf(OMEGA_CIRC * t);
97 float vy_cmd = CIRCULAR_RADIUS * OMEGA_CIRC * cosf(OMEGA_CIRC * t);
98 float angle = atan2f(vy_cmd, vx_cmd);
99 float omega_cmd = (angle - prev_angle) / DT_SECONDS;
100
101 if (omega_cmd > M_PI / DT_SECONDS) omega_cmd -= 2 * M_PI / DT_SECONDS;
102 if (omega_cmd < -M_PI / DT_SECONDS) omega_cmd += 2 * M_PI / DT_SECONDS;
103
104 prev_angle = angle;
105
106 // Write the command to the shared robot_command structure
107 if (xCmdMutex && xSemaphoreTake(xCmdMutex, pdMS_TO_TICKS(1)) == pdTRUE) {
108 robot_command.vx = vx_cmd;
109 robot_command.vy = vy_cmd;
110 robot_command.wz = omega_cmd;
111 xSemaphoreGive(xCmdMutex);
112 }
113
114 // Read the current robot state for debugging
115 if (xEstimatedDataMutex && xSemaphoreTake(xEstimatedDataMutex, pdMS_TO_TICKS(1)) == pdTRUE) {
116 speed_estimated = robot_estimated;
117 xSemaphoreGive(xEstimatedDataMutex);
118 }
119
120 // Log the command for debugging every 1s
121 if ((int)(t * 1000) % 1000 == 0) { // Log every second
122 printf("Move Task, t=%.2f, vx=%.2f, vy=%.2f, wz=%.2f\n", t, vx_cmd, vy_cmd, omega_cmd);
123 printf("Estimated Speed: vx=%.2f, vy=%.2f, wz=%.2f\n",
124 speed_estimated.vx, speed_estimated.vy, speed_estimated.wz);
125 }
126
127 // Wait for the next cycle
128 t += DT_SECONDS;
129 vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(TASK_PERIOD_MS));
130 }
131}
