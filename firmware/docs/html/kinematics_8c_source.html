40void init_forward_kinematics_matrix() {
41 float A[3][3];
42
43 // Construct the Jacobian matrix A
44 for (int i = 0; i < 3; ++i) {
45 float a = alpha[i];
46 A[i][0] = -sinf(a) / ROBOT_WHEEL_RADIUS;
47 A[i][1] = cosf(a) / ROBOT_WHEEL_RADIUS;
48 A[i][2] = ROBOT_BODY_RADIUS / ROBOT_WHEEL_RADIUS;
49 }
50
51 // Compute AtA = Aᵗ · A
52 float AtA[3][3] = {0};
53 for (int i = 0; i < 3; ++i)
54 for (int j = 0; j < 3; ++j)
55 for (int k = 0; k < 3; ++k)
56 AtA[i][j] += A[k][i] * A[k][j];
57
58 // Determinant of AtA
59 float det = AtA[0][0]*(AtA[1][1]*AtA[2][2] - AtA[1][2]*AtA[2][1])
60 - AtA[0][1]*(AtA[1][0]*AtA[2][2] - AtA[1][2]*AtA[2][0])
61 + AtA[0][2]*(AtA[1][0]*AtA[2][1] - AtA[1][1]*AtA[2][0]);
62
63 if (fabsf(det) < 1e-6f) return; // Singular matrix
64
65 float invDet = 1.0f / det;
66
67 // Compute inverse(AtA)
68 float invAtA[3][3];
69 invAtA[0][0] = (AtA[1][1]*AtA[2][2] - AtA[1][2]*AtA[2][1]) * invDet;
70 invAtA[0][1] = -(AtA[0][1]*AtA[2][2] - AtA[0][2]*AtA[2][1]) * invDet;
71 invAtA[0][2] = (AtA[0][1]*AtA[1][2] - AtA[0][2]*AtA[1][1]) * invDet;
72 invAtA[1][0] = -(AtA[1][0]*AtA[2][2] - AtA[1][2]*AtA[2][0]) * invDet;
73 invAtA[1][1] = (AtA[0][0]*AtA[2][2] - AtA[0][2]*AtA[2][0]) * invDet;
74 invAtA[1][2] = -(AtA[0][0]*AtA[1][2] - AtA[0][2]*AtA[1][0]) * invDet;
75 invAtA[2][0] = (AtA[1][0]*AtA[2][1] - AtA[1][1]*AtA[2][0]) * invDet;
76 invAtA[2][1] = -(AtA[0][0]*AtA[2][1] - AtA[0][1]*AtA[2][0]) * invDet;
77 invAtA[2][2] = (AtA[0][0]*AtA[1][1] - AtA[0][1]*AtA[1][0]) * invDet;
78
79 // Transpose A into At

82 for (int j = 0; j < 3; ++j)
