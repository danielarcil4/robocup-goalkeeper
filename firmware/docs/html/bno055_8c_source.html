711int8_t BNO055_GetCalibrationProfile(BNO055_t *bno055, BNO055_CalibProfile_t *calib_data)
712{
713 if (bno055 == NULL || calib_data == NULL) {
714 printf("Error: Null pointer provided\n");
715 return BNO055_ERROR; // Error: Null pointer
716 }
717
718 // Get the calibration status
719 BNO055_GetCalibrationStatus(bno055);
720 if (bno055->calib_stat != BNO055_CALIB_STAT_OK) {
721 printf("Error: Calibration status is not OK\n");
722 return BNO055_ERROR;
723 }
724
725 /* Save the current operation mode */
726 BNO055_OperationMode current_mode = bno055->operation_mode;
727
728 // Switch to CONFIG_MODE to read calibration profile
729 int8_t success = BNO055_SetOperationMode(bno055, CONFIGMODE);
730 if (success != BNO055_SUCCESS) {
731 printf("Error: Failed to switch to CONFIG_MODE\n");
732 return BNO055_ERROR;
733 }
734
735
736 // Read sensor offsets
737 uint8_t calib_offsets[22] = {0};
738 //success = BNO055_Read_Uart(bno055, BNO055_ACCEL_OFFSET_X_LSB_ADDR, calib_offsets, 22, 12);
739 success = BNO055_Read(bno055, BNO055_ACCEL_OFFSET_X_LSB_ADDR, calib_offsets, 22);
740 if (success != BNO055_SUCCESS) {
741 printf("Error: Failed to read calibration offsets\n");
742 // Switch back to the previous operation mode
743 BNO055_SetOperationMode(bno055, current_mode);
744 return BNO055_ERROR;
745 }
746
747 // Parse accelerometer offsets
748 calib_data->accel_offset_x = (int16_t)(calib_offsets[1] << 8 | calib_offsets[0]);
749 calib_data->accel_offset_y = (int16_t)(calib_offsets[3] << 8 | calib_offsets[2]);
750 calib_data->accel_offset_z = (int16_t)(calib_offsets[5] << 8 | calib_offsets[4]);
751
752 // Parse magnetometer offsets
753 calib_data->mag_offset_x = (int16_t)(calib_offsets[7] << 8 | calib_offsets[6]);
754 calib_data->mag_offset_y = (int16_t)(calib_offsets[9] << 8 | calib_offsets[8]);
755 calib_data->mag_offset_z = (int16_t)(calib_offsets[11] << 8 | calib_offsets[10]);
756
757 // Parse gyroscope offsets
758 calib_data->gyro_offset_x = (int16_t)(calib_offsets[13] << 8 | calib_offsets[12]);
759 calib_data->gyro_offset_y = (int16_t)(calib_offsets[15] << 8 | calib_offsets[14]);
760 calib_data->gyro_offset_z = (int16_t)(calib_offsets[17] << 8 | calib_offsets[16]);
761
762 // Parse sensor radius
763 calib_data->accel_radius = (int16_t)(calib_offsets[19] << 8 | calib_offsets[18]);
764 calib_data->mag_radius = (int16_t)(calib_offsets[21] << 8 | calib_offsets[20]);
765
766 // Switch back to the previous operation mode
767 success = BNO055_SetOperationMode(bno055, current_mode);
768 if (success != BNO055_SUCCESS) {
769 printf("Error: Failed to switch back to previous operation mode\n");
770 return BNO055_ERROR;
771 }
772
773 // Print the calibration profile HEX
774 printf("------------------------------------------------------\n");
775 printf("Calibration Profile:\n");
776 printf("Accel Offsets X: %04X\n, y: %04X\n, z: %04X\n", calib_data->accel_offset_x, calib_data->accel_offset_y, calib_data->accel_offset_z);
777 printf("Mag Offsets X: %04X\n, y: %04X\n, z: %04X\n", calib_data->mag_offset_x, calib_data->mag_offset_y, calib_data->mag_offset_z);
778 printf("Gyro Offsets X: %04X\n, y: %04X\n, z: %04X\n", calib_data->gyro_offset_x, calib_data->gyro_offset_y, calib_data->gyro_offset_z);
779 printf("Accel Radius: %04X\n", calib_data->accel_radius);
780 printf("Mag Radius: %04X\n", calib_data->mag_radius);
781 printf("------------------------------------------------------\n");
782 return BNO055_SUCCESS;
783}
