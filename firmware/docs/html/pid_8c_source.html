46static void pid_calc_positional(pid_block_t *pid)
47{
48 float output = 0;
49
50 // Anti-windup: Only integrate if the output is not saturated or if the error is driving the output back within limits
51 if ((pid->output < pid->max_output && pid->output > pid->min_output) ||
52 (pid->error > 0 && pid->output < pid->max_output) ||
53 (pid->error < 0 && pid->output > pid->min_output)) {
54 pid->integral_err += pid->error;
55 }
56
57 /* Limit the integral term */
58 // pid->integral_err = MIN(pid->integral_err, pid->max_output);
59 // pid->integral_err = MAX(pid->integral_err, pid->min_output);
60
61 /* Derivative term with beta filter */
62 pid->derivative_err = pid->error - pid->previous_err1;
63 pid->derivative_err = pid->beta * pid->previous_derivative_err + (1 - pid->beta) * pid->derivative_err;
64 /* Calculate the pid control value by location formula */
65 /* u(n) = e(n)*Kp + (e(n)-e(n-1))*Kd + integral*Ki */
66 output = pid->error * pid->Kp +
67 pid->derivative_err * pid->Kd +
68 pid->integral_err * pid->Ki;
69
70 /* If the output is out of the range, it will be limited */
71 output = MIN(output, pid->max_output);
72 output = MAX(output, pid->min_output);
73
74 /* Update previous error */
75 pid->previous_err1 = pid->error;
76 pid->previous_derivative_err = pid->derivative_err;
77
78 pid->output = output;
79}

89static void pid_calc_incremental(pid_block_t *pid)
90{
91 float output = 0;
92
93 /* Calculate the pid control value by increment formula */
94 /* du(n) = (e(n)-e(n-1))*Kp + (e(n)-2*e(n-1)+e(n-2))*Kd + e(n)*Ki */
95 /* u(n) = du(n) + u(n-1) */
96 output = (pid->error - pid->previous_err1) * pid->Kp +
97 (pid->error - 2 * pid->previous_err1 + pid->previous_err2) * pid->Kd +
98 pid->error * pid->Ki +
99 pid->previous_output;
100
101 /* If the output is beyond the range, it will be limited */
102 output = MIN(output, pid->max_output);
103 output = MAX(output, pid->min_output);
104
105 /* Update previous error */
106 pid->previous_err2 = pid->previous_err1;
107 pid->previous_err1 = pid->error;
108
109 /* Update last output */
110 pid->previous_output = output;
111
112 pid->output = output;
113}

120int pid_new_control_block(const pid_config_t *config, pid_block_handle_t *ret_pid)
121{
122 int ret = PID_OK;
123 pid_block_t *pid = NULL;
124 /* Check the input pointer */
125 if (!config || !ret_pid) {
126 ret = PID_ERR_INVALID_ARG; // Invalid argument
127 goto err;
128 }
129
130 /* Allocate memory for PID control block */
131 pid = calloc(1, sizeof(pid_block_t));
132 if (!pid) {
133 ret = PID_ERR_NO_MEM; // Out of memory
134 goto err;
135 }
136
137 /* Initialize the PID control block */
138 if (pid_update_parameters(pid, &config->init_param) != 0) {
139 goto err;
140 }
141 *ret_pid = pid;
142 return ret;
143
144err:
145 if (pid) {
146 free(pid);
147 }
148 return ret;
149}

165{

169

182int pid_update_parameters(pid_block_handle_t pid, const pid_parameter_t *params)
183{
184 if (!pid || !params) {

186 }
187
188 pid->Kp = params->kp;
189 pid->Ki = params->ki;
190 pid->Kd = params->kd;
191 pid->max_output = params->max_output;
192 pid->min_output = params->min_output;
193 pid->set_point = params->set_point;
194 pid->beta = params->beta;

196
197 /* Set the calculate function according to the PID type */
198 switch (params->cal_type) {
199 case PID_CAL_TYPE_INCREMENTAL:
200 pid->calculate_func = pid_calc_incremental;
201 break;
202 case PID_CAL_TYPE_POSITIONAL:
203 pid->calculate_func = pid_calc_positional;
204 break;
205 default:
206 return PID_ERR_INVALID_ARG; // Invalid PID calculation type
207 }
208 return PID_OK;
