112void vTaskUartHandler(void *arg) {
113 uart_event_t event;
114 char d;
115
116 while (1) {
117 if (xQueueReceive(uart_queue, (void *)&event, portMAX_DELAY)) {
118 if (event.type == UART_DATA) {
119 for (int i = 0; i < event.size; ++i) {
120 if (uart_read_bytes(UART_NUM_0, (uint8_t *)&d, 1, pdMS_TO_TICKS(10)) > 0) {
121 if (d == '\n' || d == '\r') {
122 uart_buffer[uart_buffer_index] = '\0';
123 uart_buffer_index = 0;
124 // Notify parser task
125 xTaskNotifyGive(xHandleParserTask);
126 } else if (uart_buffer_index < UART_RX_BUFFER_SIZE - 1) {
127 uart_buffer[uart_buffer_index++] = d;
128 }
129 }
130 }
131 }
132 }
133 }

146 xHandleParserTask = xTaskGetCurrentTaskHandle();

148 while (1) {

151 char *start = strstr(uart_buffer, "\"default\":\"");
152 if (start) {
153 start += strlen("\"default\":\"");
154 char *end = strchr(start, '"');
155 if (end && (end - start) < sizeof(parsed)) {
156 strncpy(parsed, start, end - start);
157 parsed[end - start] = '\0';
158
159 if (sscanf(parsed, "%f %f %f %f", &kp, &ki, &kd, &setpoint) == 4) {
160 if (xSemaphoreTake(xPidMutex, portMAX_DELAY) == pdTRUE) {
161 pid_param.kp = kp / 100.0f;
162 pid_param.ki = ki / 100.0f;
163 pid_param.kd = kd / 100.0f;
164 pid_param.set_point = setpoint;
165
166 // Update each PID controller with the new parameters
167 for (int i = 0; i < 3; i++) {
168 pid_update_parameters(pid[i], &pid_param);
169 }
170
171 xSemaphoreGive(xPidMutex);
172
173 printf("Updated PID: kp=%.2f, ki=%.2f, kd=%.2f, setpoint=%.2f\n", kp, ki, kd, setpoint);
174 } else {
175 printf("PID mutex busy.\n");
176 }
177 } else {
178 printf("Invalid format inside 'default'.\n");
179 }
180 } else {
181 printf("Value too long or malformed.\n");
182 }
183 } else {
